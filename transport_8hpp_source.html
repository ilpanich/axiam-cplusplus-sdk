17struct CaseInsensitiveLess {
18 bool operator()(const std::string& a, const std::string& b) const {
19 return lower(a) < lower(b);
20 }
21 static std::string lower(std::string s) {
22 for (char& c : s) {
23 if (c >= 'A' && c <= 'Z') c = static_cast<char>(c - 'A' + 'a');
24 }
25 return s;
26 }
27};
