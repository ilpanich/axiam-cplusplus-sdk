47inline void require_role(const std::optional<AxiamUser>& user,

64inline void require_access(Client& client, const std::optional<AxiamUser>& user,
65 const std::string& action, const std::string& resource_id,
66 std::optional<std::string> scope = std::nullopt) {
67 const AxiamUser& u = require_auth(user);
68 if (resource_id.empty()) {
69 // §11.3: unresolvable resource id is a programming error (400).
70 throw std::invalid_argument("invalid_request: unresolved resource id");
71 }
72 AccessDecision decision;
73 try {
74 decision = client.check_access(action, resource_id, std::move(scope), u.user_id);
75 } catch (const AuthzError&) {
76 throw; // server said 403/409 → denied
77 } catch (const NetworkError&) {
78 // §11.5: fail closed on transport failure; never allow.
79 throw AuthzError("authz_unavailable");
80 }
81 if (!decision.allowed) {
82 throw AuthzError("authorization_denied");
83 }
84}

100class AxiamGuard {

102 using Authenticator = std::function<std::optional<AxiamUser>(const Request&)>;

104 explicit AxiamGuard(Authenticator auth) : auth_(std::move(auth)) {}

106 AxiamUser operator()(const Request& request) const {
